1% Computes basic phonation test features.
 2% Inputs:
 3%  audio - mono floating-point, [-1,+1] normalized phonation
 4%          signal
 5%  srate - sample rate in Hz
 6%
 7% (CC BY-SA 3.0) Max Little, 2014
 8function ft = features_bvav2(audio,srate)
 9
10% Output feature vector
11ft = NaN(1,13);
12
13% Ignore zero-length inputs
14N = length(audio);
15if (N == 0)
16    return;
17end
18
19% Trim away start/end of phonation
20Tstart = 1.0;
21Tend   = 10.0;
22istart = floor(Tstart*srate);
23iend   = ceil(Tend*srate);
24if (istart > N)
25    istart = 1;
26end
27if (iend > N)
28    iend = N;
29end
30audiotrim = audio(istart:iend);
31
32N = length(audiotrim);
33T = N/srate;
34
35% F0 features
36f0dt = 0.02;
37f0lo = 50;
38f0hi = 500;
39
40% Get F0 estimates
41[f0,f0t,f0p] = swipep(audiotrim,srate,[f0lo f0hi],f0dt,NaN,NaN,NaN,NaN);
42
43% Strip away all low-probability poor F0 estimates
44i = (f0p >= 0.2);
45f0 = f0(i);
46f0t = f0t(i);
47f0p = f0p(i);
48
49% Median F0
50f0m = median(f0);
51
52% Two jitter estimates
53f0j = mean(abs(diff(f0))/f0dt);
54f0jr = median(abs(diff(f0))/f0dt);
55
56% Amplitude features
57ampdt = 0.05;
58ampwin = round(ampdt*srate);
59%abuf = buffer(audiotrim,ampwin);
60if coder.target('MATLAB')
61    abuf = buffer(audiotrim,ampwin);
62else
63    buflen = (N + ampwin - 1) / ampwin;
64    abuf = zeros(ampwin,buflen);
65    coder.ceval('buffer_nooverlap',coder.ref(abuf),audiotrim,N,ampwin);
66end
67
68l1amp = mean(abs(abuf));
69l2amp = sqrt(mean(abuf.^2));
70
71% Two shimmer estimates
72ash = mean(abs(diff(l2amp))/ampdt);
73ashr = median(abs(diff(l2amp))/ampdt);
74
75% Compute MFCCs
76cep = mfcc(audiotrim)';
77
78% MFCC summaries across time
79cepm = median(cep);
80cept = linspace(0,T,size(cep,1))';
81cepdt = cept(2)-cept(1);
82
83% MFCC jitter measure
84cepj = mean(abs(diff(cep))/cepdt);
85
86% Output phonation feature vector
87ft = [f0m f0j f0jr ash ashr cepm(1:4) cepj(1:4)];
