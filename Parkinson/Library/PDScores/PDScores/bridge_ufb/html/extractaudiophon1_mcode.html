1% Finds the best segment of a phonation recording.
 2% Inputs:
 3%  rawaudio - mono floating-point, [-1,+1] normalized phonation recording
 4% Outputs:
 5%  audiout - best phonation signal
 6%  indout  - segment index of phonation signal
 7%
 8% (CC BY-SA 3.0) Max Little, 2014
 9function [audioout, indout] = extractaudiophon(rawaudio)
10
11srate = 44100;
12mindur = floor(2.0*srate);
13
14[aud,ind] = vadsplitphon(rawaudio);
15
16audioout = [];
17indout = [];
18if (length(aud) >= 1)
19    %L = [];
20    for i = 1:length(aud)
21        L(i) = ind(i,1) - ind(i, 0) + 1;
22        %L(i) = length(aud{i});
23    end
24    [N,i] = max(L);
25
26    if (N >= mindur)
27        %audioout = aud{i};
28        audi = aud(i);
29        audioout = audi(1:L(i));
30        indout = ind(i,:);
31    end
32end
