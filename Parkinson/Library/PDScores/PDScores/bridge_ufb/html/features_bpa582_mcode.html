1% Computes basic posture test features.
 2% Inputs:
 3%  post - posture accelerometry vector: post(:,1) - time points,
 4%         post(:,2:4) - X,Y,Z acceleration data
 5%
 6% (CC BY-SA 3.0) Max Little, 2014
 7function ft = features_bpa(post)
 8
 9% Output feature vector
10ft = NaN(1,3);
11
12% Ignore zero-length inputs
13N = size(post,1);
14if (N == 0)
15    return;
16end
17
18% Calculate relative time
19t = post(:,1)-post(1,1);
20
21Tstart = 3.0;
22Tend   = 19.0;
23
24% Ignore posture tests which do not contain enough data
25if (max(t) < Tend)
26    return;
27end
28
29% Trim away start/end of test
30istart = find(t >= Tstart,1,'first');
31iend   = find(t <= Tend,1,'last');
32
33if (iend(1) < istart(1))
34    return;
35end
36
37posttrim = post(istart(1):iend(1),2:4);
38ttrim = t(istart(1):iend(1));
39
40N = length(posttrim);
41T = ttrim(end);
42dT = T-ttrim(1);
43
44% Orientation
45mg = mean(posttrim);
46
47% Orientation-corrected force signals
48postforce = posttrim-repmat(mg,N,1);
49
50% Scaled velocity signals
51dt = diff(ttrim);
52%dt(end+1) = dt(end); -- causes code gen bug (doesn't update size of dt)
53dt = [dt; dt(end)];
54postvel = cumsum(postforce.*repmat(dt,1,3));
55
56% Average scaled power X,Y,Z
57postpower = mean(sum(0.5*70*postvel.^2)/dT)/1e4;
58
59% Force vector magnitude signal
60postmag = sqrt(sum(postforce.^2,2));
61
62% Maximum force
63postpeak = quantile(postmag,0.95)/10;
64
65% Detrended fluctuation analysis scaling exponent
66[alpha,iv,fl] = fastdfa(postmag);
67
68% Output posture test feature vector
69ft = [postpeak postpower alpha];
