1% Segment a phonation recording, based on an attack-sustain-release state
  2% machine model.
  3% Inputs:
  4%  x - mono floating-point, [-1,+1] normalized phonation recording
  5% Outputs:
  6%  y - phonation segments
  7%  i - sample index of start of each segment
  8%
  9% (CC BY-SA 3.0) Max Little, 2014
 10function [y,i] = vadsplitphon(x)
 11
 12% Normalize audio input
 13x = x/max(abs(x));
 14
 15srate = 44100;              % Sample rate
 16framedur = 0.04;            % Frame duration (sec)
 17framechange = 0.01;         % Frame change duration (sec)
 18maxsilencedur = 0.5;        % Post-voice activity silence dur (sec)
 19ampthresh = 0.05;           % Voice threshold
 20
 21framesize  = ceil(framedur*srate);
 22frameinc   = ceil(framechange*srate);
 23maxsilenceframes = ceil(maxsilencedur/framechange);
 24
 25% Mean absolute amplitude
 26fskip = ceil(framesize/frameinc);   % Ignore first few incomplete frames
 27if coder.target('MATLAB')
 28    X = buffer(x,framesize,framesize-frameinc);
 29else
 30    buflen = (length(x) + frameinc - 1) / frameinc;
 31    X = zeros(framesize, buflen);
 32    coder.ceval('buffer_overlap',coder.ref(X),x,length(x),framesize,framesize-frameinc);
 33end
 34amp = mean(abs(X(:,fskip:end)))';
 35N = length(amp);
 36
 37% Loop pre-conditions
 38n = 1;
 39k = 0;
 40y = [];
 41i = [];
 42coder.varsize('i');
 43state = 0;  % 0-silent wait for voice, 1-voice active, 2-silent sustain
 44activeframes = 0;
 45silenceframes = 0;
 46tstart = 0;
 47
 48while (n <= N)
 49    
 50    switch (state)
 51
 52        % Entry state: waiting for voice activity.
 53        case 0,
 54            if (amp(n) > ampthresh)
 55                activeframes = 1;
 56                tstart = n;
 57                state = 1;
 58            end
 59
 60        % Tracking voice activity
 61        case 1,
 62            if (amp(n) > ampthresh)
 63                % Continue counting frames exceeding threshold
 64                activeframes = activeframes + 1;
 65            else
 66                % Amplitude below threshold: begin silence sustain
 67                activeframes = activeframes + 1;
 68                silenceframes = 1;
 69                state = 2;
 70            end
 71
 72        % Silence sustain phase
 73        case 2,
 74            if (amp(n) <= ampthresh)
 75                % Amplitude below threshold: continue sustain
 76                silenceframes = silenceframes + 1;
 77                activeframes = activeframes + 1;
 78                
 79                if (silenceframes >= maxsilenceframes)
 80                    % Exceeded max silence sustain duration, capture return
 81                    % to wait for next activity.
 82                    tend = tstart + activeframes - 1;
 83                    i0 = (tstart-1)*frameinc+1;
 84                    i1 = tend*frameinc;
 85                    k = k + 1;
 86                    y(k) = x(i0:i1);
 87                    i(k,1) = i0;
 88                    i(k,2) = i1;
 89                    state = 0;
 90                end
 91            else
 92                % Ampltude goes above threshold again, cancel sustain
 93                activeframes = activeframes + 1;
 94                state = 1;
 95            end
 96
 97    end
 98    n = n + 1;
 99    
100end
101
102% Deal with ending in non-entry state
103if ((state == 1) || (state == 2))
104    tend = tstart + activeframes - 1;
105    i0 = (tstart-1)*frameinc+1;
106    i1 = tend*frameinc;
107    k = k + 1;
108    y(k) = x(i0:i1);
109    i(k,1) = i0;
110    i(k,2) = i1;
111end
